from sklearn.tree import DecisionTreeRegressor

def cross_val_score(estimator, X, y, scoring=performance_metric, cv=3):
    """  """
    n1 = len(X)/3
    n2 = n1*2
    X1 = X[:n1]
    y1 = y[:n1]
    X2 = X[n1:n2]
    y2 = y[n1:n2]
    X3 = X[n3:]
    y3 = y[n3:]
    reg1 = estimator.fit(np.append(X2,X3,axis=0), np.append(y2,y3, axis=0))
    pred1 = reg1.predict(X1)
    
    reg2 = estimator.fit(np.append(X1,X3,axis=0), np.append(y1,y3,axis=0))
    pred2 = reg2.predict(X2)
    
    reg3 = estimator.fit(np.append(X1,X2,axis=0), np.append(y1,y2,axis=0))
    pred3 = reg3.predict(X3)
    
    scores = [scoring(y1,pred1), scoring(y2,pred2), scoring(y3, pred3)]
    
    
    return scores

def fit_model2(X, y):
    """  [X,y]"""
    
    #
    compareDic = {}
    for depth in range(1,11):
        regressor = DecisionTreeRegressor(max_depth=depth)
        score = cross_val_score(regressor, X, y, scoring=performance_metric, cv=3)
        compareDic[depth] = np.mean(score)
        
    print compareDic
    best_score = max(compareDic.items(), key=lambda x:x[1])
    
    
    best_estimator = DecisionTreeRegressor(max_depth=best_score[0])
    
    return best_estimator
